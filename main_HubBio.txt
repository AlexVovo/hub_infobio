void main() async {
  WidgetsFlutterBinding.ensureInitialized();

  runApp(const BioHubApp());
}

class BioHubApp extends StatefulWidget {
  const BioHubApp({super.key});

  @override
  State<BioHubApp> createState() => _BioHubAppState();
}

class _BioHubAppState extends State<BioHubApp> {
  ThemeMode _themeMode = ThemeMode.system; // modo inicial autom√°tico

  void _toggleTheme() {
    setState(() {
      if (_themeMode == ThemeMode.light) {
        _themeMode = ThemeMode.dark;
      } else if (_themeMode == ThemeMode.dark) {
        _themeMode = ThemeMode.system;
      } else {
        _themeMode = ThemeMode.light;
      }
    });
  }

  IconData get _themeIcon {
    switch (_themeMode) {
      case ThemeMode.light:
        return Icons.dark_mode;
      case ThemeMode.dark:
        return Icons.brightness_auto;
      default:
        return Icons.light_mode;
    }
  }

  String get _themeTooltip {
    switch (_themeMode) {
      case ThemeMode.light:
        return 'Modo escuro';
      case ThemeMode.dark:
        return 'Modo autom√°tico';
      default:
        return 'Modo claro';
    }
  }

  @override
  Widget build(BuildContext context) {
    return AnimatedTheme(
      // ‚ú® Anima√ß√£o de transi√ß√£o suave entre temas
      data: ThemeData(
        colorScheme: ColorScheme.fromSeed(seedColor: Colors.teal),
      ),
      duration: const Duration(milliseconds: 500), // tempo da anima√ß√£o
      curve: Curves.easeInOut,
      child: MaterialApp(
        title: 'Hub de Bioinform√°tica',
        debugShowCheckedModeBanner: false,
        themeMode: _themeMode,
        theme: ThemeData(
          colorScheme: ColorScheme.fromSeed(seedColor: Colors.teal),
          useMaterial3: true,
          brightness: Brightness.light,
          visualDensity: VisualDensity.adaptivePlatformDensity,
        ),
        darkTheme: ThemeData(
          colorScheme: ColorScheme.fromSeed(
            seedColor: Colors.teal,
            brightness: Brightness.dark,
          ),
          useMaterial3: true,
          visualDensity: VisualDensity.adaptivePlatformDensity,
        ),
        home: AnimatedSwitcher(
          duration: const Duration(milliseconds: 400),
          transitionBuilder: (child, animation) =>
              FadeTransition(opacity: animation, child: child),
          child: HomePage(
            key: ValueKey(_themeMode), // garante que o fade funcione
            onToggleTheme: _toggleTheme,
            themeIcon: _themeIcon,
            themeTooltip: _themeTooltip,
          ),
        ),
      ),
    );
  }
}
